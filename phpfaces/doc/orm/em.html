Entity Management in Entity's class . Save and Delete methods

 

class Comment extends Entity{
/**
@Id
@Column(name = "id" ,type = "integer")
*/
private $id;
/**
@Column(name = "content" ,type = "text")
*/
private $content;
public function set($name, $value) {
$this->$name= $value;
}
public function get($name) {
return $this->$name;
}
function save() {
$em = Manager::getInstance();
$em->beginTransaction();
try {
$em->save($this);
}
$em->commit();
} catch (Exception $e) {
$em->rollBack();
}
}
function delete() {
$em = Manager::getInstance();
$em->beginTransaction();
try {
$em->delete($this);
}
$em->commit();
} catch (Exception $e) {
$em->rollBack();
}
}}
